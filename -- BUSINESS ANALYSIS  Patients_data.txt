-- Data Analysics: Patients_data

-- Question 1: Calculate readmission counts based on missing follow-up scheduled?
SELECT 
    follow_up_scheduled,
    COUNT(patient_id) AS total_patients,
    SUM(CASE WHEN readmitted_30_days = 'Yes' THEN 1 ELSE 0 END) AS total_readmissions
FROM 
     Patients_data
GROUP BY
     follow_up_scheduled

-- Question 2: Which diagnosis group has more patients with age?

SELECT 
      diagnosis_group,
      COUNT(patients_id) as total_patients,
      AVG(age) AS average_age
FROM
      Patients_data
GROUP BY
      diagnosis_group
ORDER BY
      average_age DESC
